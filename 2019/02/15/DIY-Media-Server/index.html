DIY Media Server

DIY Media Server

##路由主板
需要一块工控路由主板，可到Taobao, JD上搜索购买。

我买的主板配置：

  • CPU :i5-2540M。
  • 内存 :8G。
  • SSD :120G。

##制作安装U盘
需要一个4G以上的U盘。

  1. 下载安装镜像。

  1. 格式化U盘，并把系统写入U盘。
    Mac 可以执行：

vi /etc/sysconfig/network-scripts/ifcfg-eth0

TYPE=Ethernet
ONBOOT=yes
NM_CONTROLLED=yes
BOOTPROTO=dhcp

yum install ppp rp-pppoe

pppoe-setup

vi /etc/sysconfig/network-scripts/ifcfg-br0

DEVICE=br0
TYPE=Bridge
BOOTPROTO=static
IPADDR=192.168.1.1
NETMASK=255.255.255.0
ONBOOT=yes
NM_CONTROLLED=no
PREFIX=24
ARPCHECK=no

vi /etc/sysconfig/network-scripts/ifcfg-eth1

ONBOOT=yes
NM_CONTROLLED=yes
BOOTPROTO=none
BRIDGE=br0
ARPCHECK=no

yum install dnsmasq

cp /etc/hosts /etc/dnsmasq.d/hosts
echo ‘127.0.0.1 localhost centos.router’ > /etc/dnsmasq.d/hosts

cp /etc/resolv.conf /etc/dnsmasq.d/resolv.conf
echo ‘nameserver 8.8.8.8’ > /etc/dnsmasq.d/resolv.conf

echo “conf-dir=/etc/dnsmasq.d” >> /etc/dnsmasq.conf

wget -4 –no-check-certificate -O /etc/dnsmasq.d/accelerated-domains.china.conf https://github.com/felixonmars/dnsmasq-china-list/raw/master/accelerated-domains.china.conf

wget -4 –no-check-certificate -O /etc/dnsmasq.d/bogus-nxdomain.china.conf https://github.com/felixonmars/dnsmasq-china-list/raw/master/bogus-nxdomain.china.conf

vi /etc/dnsmasq.conf

log-queries
log-facility=/var/log/dnsmasq.log
log-async=20

addn-hosts=/etc/dnsmasq.d/hosts
no-hosts

resolv-file=/etc/dnsmasq.d/resolv.conf
strict-order
no-resolv
no-poll

dhcp-range=192.168.1.50,192.168.1.150,255.255.255.0,12h
dhcp-option=192.168.1.1
domain=centos.router

service dnsmasq restart

wget -O- ‘http://ftp.apnic.net/apnic/stats/apnic/delegated-apnic-latest' | awk -F| ‘/CN|ipv4/ { printf(“%s/%d\n”, $4, 32-log($5)/log(2)) }’ > /etc/chinadns_chnroute.txt

cd /etc/yum.repos.d/
wget https://copr.fedoraproject.org/coprs/librehat/shadowsocks/repo/epel-6/librehat-shadowsocks-epel-6.repo

yum update
yum install epel-release -y
yum install shadowsocks-libev

vim /etc/sysctl.conf

net.ipv4.ip_forward = 1

vim router.firewall.sh

chmod +x router.firewall.sh

#!/bin/bash

#Your SS Server IP
SHADOWSOCKS_SERVER_HOST=”x.x.x.x”

#Your SS Server Local Port
SHADOWSOCKS_LOCAL_PORT=”xxxx”

sed -i ‘s/114.114.114.114/127.0.0.1#3210/‘ /etc/dnsmasq.d/gfwlist.conf
/etc/init.d/dnsmasq restart

/usr/bin/ss-redir stop
/usr/bin/ss-tunnel stop

killall ss-redir
killall ss-tunnel

echo “server=/#/127.0.0.1#3210” > /etc/dnsmasq.d/gfwlist.conf

nohup /usr/bin/ss-redir -b 0.0.0.0 -c /etc/shadowsocks.json -f /var/run/shadowsocks.pid -u -v &
nohup /usr/bin/ss-tunnel -b 0.0.0.0 -c /etc/shadowsocks.json -l 3210 -L 8.8.8.8:53 -u -v &

#INPUT
iptables -F
iptables -X
iptables -Z

iptables -P INPUT DROP
iptables -P OUTPUT ACCEPT
iptables -P FORWARD ACCEPT

iptables -A INPUT -i lo -j ACCEPT
iptables -A INPUT -m state –state RELATED,ESTABLISHED -j ACCEPT

iptables -A INPUT -d 0.0.0.0/8 -j ACCEPT
iptables -A INPUT -d 10.0.0.0/8 -j ACCEPT
iptables -A INPUT -d 127.0.0.0/8 -j ACCEPT
iptables -A INPUT -d 169.254.0.0/16 -j ACCEPT
iptables -A INPUT -d 172.16.0.0/12 -j ACCEPT
iptables -A INPUT -d 192.168.0.0/16 -j ACCEPT
iptables -A INPUT -d 224.0.0.0/4 -j ACCEPT
iptables -A INPUT -d 240.0.0.0/4 -j ACCEPT

#NAT
iptables -F -t nat
iptables -X -t nat
iptables -Z -t nat

iptables -t nat -N SHADOWSOCKS
iptables -t nat -N SHADOWSOCKS_WHITELIST

iptables -t nat -A SHADOWSOCKS -d $SHADOWSOCKS_SERVER_HOST -j RETURN
iptables -t nat -A SHADOWSOCKS -d 0.0.0.0/8 -j RETURN
iptables -t nat -A SHADOWSOCKS -d 10.0.0.0/8 -j RETURN
iptables -t nat -A SHADOWSOCKS -d 127.0.0.0/8 -j RETURN
iptables -t nat -A SHADOWSOCKS -d 169.254.0.0/16 -j RETURN
iptables -t nat -A SHADOWSOCKS -d 172.16.0.0/12 -j RETURN
iptables -t nat -A SHADOWSOCKS -d 192.168.0.0/16 -j RETURN
iptables -t nat -A SHADOWSOCKS -d 224.0.0.0/4 -j RETURN
iptables -t nat -A SHADOWSOCKS -d 240.0.0.0/4 -j RETURN

iptables -t nat -A SHADOWSOCKS -j SHADOWSOCKS_WHITELIST
iptables -t nat -A SHADOWSOCKS -m mark –mark 1 -j RETURN

iptables -t nat -A SHADOWSOCKS -p tcp -j REDIRECT –to-ports $SHADOWSOCKS_LOCAL_PORT
iptables -t nat -A PREROUTING -p tcp -j SHADOWSOCKS

for white_ip in cat /etc/chinadns_chnroute.txt;
do
iptables -t nat -A SHADOWSOCKS_WHITELIST -d “${white_ip}” -j MARK –set-mark 1
echo ${white_ip}
done

iptables -t nat -A POSTROUTING -o eth0 -s 192.168.1.0/24 -j MASQUERADE
iptables -A FORWARD -s 192.168.1.0/24 -j ACCEPT
iptables -A FORWARD -d 192.168.1.0/24 -j ACCEPT

/etc/init.d/iptables save

./router.firewall.sh
~~~

EOF
